public class htmlUsuarios {

    public String header = "<!DOCTYPE html>\n"
            + "<html>\n"
            + "<head>\n"
            + "	<meta charset=\"utf-8\">\n"
            + "	<meta name=\"viewport\" content=\"width=device-width, initial-scale=1\">\n"
            + "	<title>REPORTE Numero4</title>\n"
            + "</head>\n"
            + "<body>\n"
            + "	<div class=\"Nombre\">\n"
            + "		<style type=\"text/css\">\n"
            + "			p{\n"
            + "				text-align: center;\n"
            + "			}\n"
            + "		</style>\n"
            + "		<h1><p>Andrés Emilio Peñate Hernández</p></h1>\n"
            + "		<h2><p>201708550</p></h2>\n"
            + "	</div>\n"
            + "	<style type=\"text/css\">\n"
            + "		.tftable {font-size:12px;color:#333333;width:100%;border-width: 1px;border-color: #729ea5;border-collapse: collapse;}\n"
            + "		.tftable th {font-size:12px;background-color:#acc8cc;border-width: 1px;padding: 8px;border-style: solid;border-color: #729ea5;text-align:left;}\n"
            + "		.tftable tr {background-color:#d4e3e5;}\n"
            + "		.tftable td {font-size:12px;border-width: 1px;padding: 8px;border-style: solid;border-color: #729ea5;}\n"
            + "		.tftable tr:hover {background-color:#ffffff;}\n"
            + "	</style>\n"
            + "\n"
            + "	<table class=\"tftable\" border=\"1\">";

    public String footer = "	</table>\n"
            + "</body>\n"
            + "</html>";

    private Usuarios[] users;
    private String[] columnas;

    public htmlUsuarios(Usuarios[] users, String[] columnas) {
        this.users = users;
        this.columnas = columnas;
    }

    public String obtenerhtmlUsuarios() {
        String tablaColumnas = "";

        tablaColumnas += "        <tr>\n";

        for (int i = 0; i < columnas.length; i++) {
            tablaColumnas += "            <th>" + columnas[i] + "</th>\n";
        }
        tablaColumnas += "        </tr>\n";
        String tablaDatos = "";
        for (int i = 0; i < users.length; i++) {
            if (users[i] != null) {
                tablaDatos += "        <tr>\n";

                tablaDatos += "            <td>" + users[i].getid() + "</td>\n";
                tablaDatos += "            <td>" + users[i].getnombre() + "</td>\n";
                tablaDatos += "            <td>" + users[i].getapellido() + "</td>\n";
                tablaDatos += "            <td>" + users[i].getusuario() + "</td>\n";
                tablaDatos += "            <td>" + users[i].getrol() + "</td>\n";

                tablaDatos += "        </tr>\n";
            }
        }

        return header + tablaColumnas + tablaDatos + footer;
    }
}
